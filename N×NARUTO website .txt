<script>
const ADMIN_WHATSAPP = "94758838871";

const DEFAULT_PACKS = {
diamond:[
{name:'Weekly Lite',price:130,bonus:'60 diamonds daily × 7 days',tag:''},
{name:'Weekly',price:530,bonus:'450 diamonds + Bonus rewards',tag:'HOT'},
{name:'Monthly',price:2550,bonus:'2200 diamonds + Monthly bonus',tag:''},
{name:'VIP',price:2940,bonus:'2500 diamonds + VIP perks',tag:'VIP'},
{name:'S.VIP',price:4950,bonus:'5000 diamonds + Super VIP rewards',tag:'BEST'}
],
shell:[
{name:'33 Shells',price:200,bonus:'= 100 Diamonds',tag:''},
{name:'66 Shells',price:400,bonus:'= 210 Diamonds',tag:'HOT'},
{name:'165 Shells',price:1000,bonus:'= 530 Diamonds',tag:''},
{name:'330 Shells',price:2000,bonus:'= 1080 Diamonds',tag:'BEST'},
{name:'660 Shells',price:4000,bonus:'= 2200 Diamonds',tag:'VIP'}
]};

let allPacks = {};
let selectedPack = null;
let receiptUploaded = false;

function showDebug(msg){
let box = document.getElementById('debugBox');
box.innerText = msg;
box.classList.remove('hide');
console.log(msg);
}

function initPacks(){
try{
let saved = localStorage.getItem('nxn_packs');
if(saved){
allPacks = JSON.parse(saved);
showDebug('Loaded packs from localStorage');
}else{
allPacks = DEFAULT_PACKS;
localStorage.setItem('nxn_packs', JSON.stringify(DEFAULT_PACKS));
showDebug('Using default packs - saved to localStorage');
}
renderPacks();
}catch(e){
showDebug('ERROR: ' + e.message + ' - Using defaults');
allPacks = DEFAULT_PACKS;
renderPacks();
}
}

function renderPacks(){
try{
let html = '';
let diamondPacks = allPacks.diamond || [];
let shellPacks = allPacks.shell || [];
let totalPacks = diamondPacks.length + shellPacks.length;

if(totalPacks === 0){
showDebug('ERROR: 0 packs found in data');
return;
}

if(diamondPacks.length > 0){
html += `<div class="category-title">💎 FREE FIRE DIAMONDS</div><div class="pack-grid">`;
diamondPacks.forEach((p, i) => {
html += `<div class="pack" data-type="diamond" data-index="${i}">
${p.tag?`<div class="tag">${p.tag}</div>`:''}
<div class="pack-name">${p.name}</div>
<div class="pack-price">Rs ${p.price}</div>
<div class="pack-bonus">${p.bonus}</div>
<div class="pack-type">DIAMOND</div>
</div>`;
});
html += `</div>`;
}

if(shellPacks.length > 0){
html += `<div class="category-title">🎮 GARENA SHELLS</div><div class="pack-grid">`;
shellPacks.forEach((p, i) => {
html += `<div class="pack" data-type="shell" data-index="${i}">
${p.tag?`<div class="tag">${p.tag}</div>`:''}
<div class="pack-name">${p.name}</div>
<div class="pack-price">Rs ${p.price}</div>
<div class="pack-bonus">${p.bonus}</div>
<div class="pack-type">SHELL</div>
</div>`;
});
html += `</div>`;
}

document.getElementById('packList').innerHTML = html;

document.querySelectorAll('.pack').forEach(el => {
el.addEventListener('click', function(){
selectPack(this);
});
});

showDebug('Rendered ' + totalPacks + ' packs successfully');
}catch(e){
showDebug('RENDER ERROR: ' + e.message);
}
}

function selectPack(el){
document.querySelectorAll('.pack').forEach(x=>x.classList.remove('active'));
el.classList.add('active');
let type = el.getAttribute('data-type');
let idx = parseInt(el.getAttribute('data-index'));
selectedPack = allPacks[type][idx];
selectedPack.type = type;
let typeText = type==='diamond'?'Diamond':'Shell';
document.getElementById('selectedPack').innerText = `${selectedPack.name} - Rs ${selectedPack.price} (${typeText})`;
checkForm();
}

function startVerify(){
let uid = document.getElementById('uid').value.trim();
if(uid.length < 5){ alert('Enter valid UID/Email'); return; }
window.playerUID = uid;
document.getElementById('verifyStage').classList.add('show');
document.getElementById('verifyBtn').innerText = 'ENTER NAME';
document.getElementById('verifyBtn').disabled = true;
}

function confirmName(){
let name = document.getElementById('gameName').value.trim();
if(name.length < 3){ alert('Enter your real in-game name'); return; }
window.playerName = name;
window.verified = true;
document.getElementById('displayName').innerText = window.playerName;
document.getElementById('displayUID').innerText = window.playerUID;
document.getElementById('verifiedBox').classList.add('show');
document.getElementById('verifyStage').classList.remove('show');
document.getElementById('verifyBtn').innerText = '✓ VERIFIED';
document.getElementById('verifyBtn').classList.add('verified');
checkForm();
}

function resetVerify(){
window.verified = false;
document.getElementById('verifiedBox').classList.remove('show');
document.getElementById('verifyStage').classList.remove('show');
let btn = document.getElementById('verifyBtn');
btn.innerText = 'VERIFY'; btn.classList.remove('verified'); btn.disabled = false;
checkForm();
}

function handleFileUpload(e){
let file = e.target.files[0];
if(file){
receiptUploaded = true;
document.getElementById('uploadBox').classList.add('has-file');
document.getElementById('uploadContent').innerHTML = `
<div class="upload-icon">✓</div>
<div class="upload-text">Receipt Uploaded</div>
<div class="file-name">${file.name}</div>
`;
checkForm();
}
}

function checkForm(){
let checked = document.getElementById('confirmCheck').checked;
document.getElementById('submitBtn').disabled =!(window.playerUID && window.playerUID.length >= 5 && window.verified && checked && selectedPack && receiptUploaded);
}

function submitOrder(){
let orderId = 'NXN' + Date.now();
let typeText = selectedPack.type==='diamond'?'Diamonds':'Garena Shells';

let order = {id:orderId,type:selectedPack.type,uid:window.playerUID,name:window.playerName,pack:selectedPack.name,amount:selectedPack.price,status:'Pending',time:new Date().toLocaleString()};
let orders = JSON.parse(localStorage.getItem('nxn_orders')||'[]');
orders.unshift(order);
localStorage.setItem('nxn_orders',JSON.stringify(orders));

let msg = `*NEW N×NARUTO ORDER* 🔥%0A%0A` +
`*Order ID:* ${orderId}%0A` +
`*Type:* ${typeText}%0A` +
`*Player:* ${window.playerName}%0A` +
`*UID:* ${window.playerUID}%0A` +
`*Pack:* ${selectedPack.name}%0A` +
`*Amount:* Rs ${selectedPack.price}%0A` +
`*Time:* ${new Date().toLocaleString()}%0A%0A` +
`Receipt screenshot attached. Please verify and send.`;

let waUrl = `https://wa.me/${ADMIN_WHATSAPP}?text=${msg}`;

document.getElementById('o_id').innerText = orderId;
document.getElementById('o_type').innerText = typeText;
document.getElementById('o_uid').innerText = window.playerUID;
document.getElementById('o_name').innerText = window.playerName;
document.getElementById('o_pack').innerText = selectedPack.name;
document.getElementById('o_amt').innerText = selectedPack.price;
document.getElementById('waLink').href = waUrl;
document.getElementById('orderForm').classList.add('hide');
document.getElementById('successPage').classList.remove('hide');
window.scrollTo(0,0);

setTimeout(()=>{ window.open(waUrl, '_blank'); }, 1000);
}

document.addEventListener('DOMContentLoaded', initPacks);
